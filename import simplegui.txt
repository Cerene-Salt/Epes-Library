import simplegui

# URLs das imagens
urls = [
    "https://i.imgur.com/Ioqgatq.jpeg",
    "https://i.imgur.com/YtOZuIx.jpeg",
    "https://i.imgur.com/Nn7zfoV.jpeg"
]

# Carregando as imagens
images = [simplegui.load_image(url) for url in urls]

def draw(canvas):
    y_pos = 100  # posição vertical inicial para desenhar as imagens
    
    for image in images:
        if image.get_width() > 0 and image.get_height() > 0:
            source_center = [image.get_width() / 2, image.get_height() / 2]
            source_size = [image.get_width(), image.get_height()]
            dest_center = [150, y_pos]
            dest_size = [300, 200]  # tamanho para todas as imagens
            canvas.draw_image(image, source_center, source_size, dest_center, dest_size)
        else:
            canvas.draw_text("Loading image...", [50, y_pos], 24, "Red")
        
        y_pos += 220  # espaço para próxima imagem

frame = simplegui.create_frame("Multiple Images Demo", 300, 700)
frame.set_draw_handler(draw)
frame.start()
